#include <iostream>

char space[3][3]={{'1','2','3'},{'4','5','6'},{'7','8','9'}};
string n1="";
string n2="";
int row;
int column;
char token='x';
bool tie=false;

	cout<<"\t\t\t   |     |   \n";

void functionone(){
	int digit;
	
	if(token=='x'){
		cout<<n1<<" please enter: ";
		cin>>digit;
	}
	if(token=='0'){
		cout<<n2<<" please enter: ";
		cin>>digit;
	}
	//Now we check the place of the value
	 if(digit==1){
	 	row=0;
	 	column=0;
	 }
	 if(digit==2){
	 	row=0;
	 	column=1;
	 }
	 if(digit==3){
	 	row=0;
	 	column=2;
	 }
	 if(digit==4){
	 	row=1;
	 	column=0;
	 }
	 if(digit==5){
	 	row=1;
	 	column=1;
	 }
	 if(digit==6){
	 	row=1;
	 	column=2;
	 }
	 if(digit==7){
	 	row=2;
	 	column=0;
	 }
	 if(digit==8){
	 	row=2;
	 	column=1;
	 }
	 if(digit==9){
	 	row=2;
	 	column=2;
	 }
	 else if(digit<1 || digit>9){
	 	cout<<"Invalid !!!"<<endl;
	 }
	 //Now check is there any empty space left or not
	 
	 if(token=='x' && space[row][column] !='x' && space[row][column] !='0')
	 {
	 space[row][column]='x';
	 token='0';
	 }
	 
	 else if(token=='0' && space[row][column] !='x' && space[row][column] !='0')
	 {
	 space[row][column]='0';
	 token='x';
     }
     
     else{
     	cout<<"There is no empty space left! "<<endl;
     	functionone();
	 }
}
bool functionthree(){ //now check which player is win the game or whether the game is still going on or not
   	
   	for(int i=0;i<3;i++){ //check for horizontal and vertical
   	
   	if(space[i][0]==space[i][1] && space[i][0]==space[i][2] || space[0][i]==space[1][i] && space[0][i]==space[2][i])
	{
	    return true;
	}}
	//check for diagonal 
	if(space[0][0]==space[1][1] && space[1][1]==space[2][2] || space[0][2]==space[1][1] && space[1][1]==space[2][0])
	{
	    return true;
    }
	//check condition whether the game is still going on or not
	for(int i=0;i<3;i++){
		for(int j=0;j<3;j++){
			if(space[i][j]!='x' && space[i][j]!='0'){
		 	return false;
		    }	
		}
	}
	//now check condition for tie
	tie=true;
	return false;
}
